import { TRPCError } from '@trpc/server';

// Error codes matching API-12 contract
export const ErrorCode = {
  AUTH_EXPIRED: 'AUTH_EXPIRED',
  INVALID_CREDENTIALS: 'INVALID_CREDENTIALS',
  EMAIL_EXISTS: 'EMAIL_EXISTS',
  RATE_LIMITED: 'RATE_LIMITED',
  HISTORY_NOT_FOUND: 'HISTORY_NOT_FOUND',
  SHARE_NOT_FOUND: 'SHARE_NOT_FOUND',
  SHARE_EXPIRED: 'SHARE_EXPIRED',
  VERSION_OUTDATED: 'VERSION_OUTDATED',
  SERVER_ERROR: 'SERVER_ERROR',
} as const;

export type ErrorCode = typeof ErrorCode[keyof typeof ErrorCode];

export function createAppError(
  code: keyof typeof ErrorCode,
  trpcCode: TRPCError['code'],
  details?: Record<string, unknown>
): TRPCError {
  return new TRPCError({
    code: trpcCode,
    message: ErrorCode[code],
    cause: details ? JSON.stringify(details) : undefined,
  });
}
```

Update all routers to use `createAppError()` instead of raw `new TRPCError()`.

import { buildServer } from '../server.js';

export async function createTestServer() {
  const server = await buildServer();
  return server;
}

export async function callTrpc(server: any, path: string, input?: unknown, headers?: Record<string, string>) {
  const method = input ? 'POST' : 'GET';
  const url = input
    ? `/trpc/${path}`
    : `/trpc/${path}${input !== undefined ? `?input=${encodeURIComponent(JSON.stringify(input))}` : ''}`;

  return server.inject({
    method,
    url,
    payload: input ? { json: input } : undefined,
    headers: {
      'content-type': 'application/json',
      ...headers,
    },
  });
}
```
