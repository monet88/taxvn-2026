Create `apps/api/src/lib/tokens.ts`:
```typescript
import jwt from 'jsonwebtoken';
import crypto from 'node:crypto';
import { env } from '../env.js';

const ACCESS_TOKEN_EXPIRY = '15m';
const REFRESH_TOKEN_EXPIRY = '7d';

export interface AccessTokenPayload {
  userId: string;
  email: string;
}

export function generateAccessToken(payload: AccessTokenPayload): string {
  return jwt.sign(payload, env.JWT_ACCESS_SECRET, { expiresIn: ACCESS_TOKEN_EXPIRY });
}

export function verifyAccessToken(token: string): AccessTokenPayload {
  return jwt.verify(token, env.JWT_ACCESS_SECRET) as AccessTokenPayload;
}

export function generateRefreshToken(): string {
  return crypto.randomBytes(48).toString('base64url');
}

export function hashToken(token: string): string {
  return crypto.createHash('sha256').update(token).digest('hex');
}

export function getRefreshTokenExpiry(): Date {
  const now = new Date();
  now.setDate(now.getDate() + 7);
  return now;
}
```

Create `apps/api/src/middleware/auth.ts`:
```typescript
import { TRPCError } from '@trpc/server';
import { middleware } from '../trpc.js';
import { verifyAccessToken } from '../lib/tokens.js';

export const isAuthenticated = middleware(async ({ ctx, next }) => {
  const authHeader = ctx.req.headers.authorization;
  if (!authHeader?.startsWith('Bearer ')) {
    throw new TRPCError({ code: 'UNAUTHORIZED', message: 'AUTH_EXPIRED' });
  }

  const token = authHeader.slice(7);
  try {
    const payload = verifyAccessToken(token);
    return next({ ctx: { ...ctx, user: payload } });
  } catch {
    throw new TRPCError({ code: 'UNAUTHORIZED', message: 'AUTH_EXPIRED' });
  }
});
```

<task id="02-04" title="Auth router (register, login, refresh, logout)">

<action>
Create `apps/api/src/routers/auth.ts` with tRPC procedures:

**register**: publicProcedure
- Input: `z.object({ email: z.string().email(), password: z.string().min(8), name: z.string().optional() })`
- Check if email exists → throw `CONFLICT` with message `EMAIL_EXISTS`
- Hash password with `hashPassword()`
- Create user in DB
- Generate access + refresh tokens
- Create session with hashed refresh token + device info from `req.headers['user-agent']`
- Return `{ accessToken, refreshToken, user: { id, email, name } }`

**login**: publicProcedure
- Input: `z.object({ email: z.string().email(), password: z.string() })`
- Find user by email → throw `NOT_FOUND` with message `INVALID_CREDENTIALS`
- Verify password → throw `UNAUTHORIZED` with message `INVALID_CREDENTIALS`
- Generate access + refresh tokens
- Create session
- Return same shape as register

**refresh**: publicProcedure
- Input: `z.object({ refreshToken: z.string() })`
- Hash incoming token, find matching session
- If not found or expired → throw `UNAUTHORIZED` with message `AUTH_EXPIRED`
- Delete old session (rotation: old token invalidated)
- Generate new access + refresh tokens
- Create new session
- Return `{ accessToken, refreshToken }`

**logout**: protectedProcedure
- Input: `z.object({ refreshToken: z.string() })`
- Hash token, delete session matching hash + userId
- Return `{ success: true }`

Create `apps/api/src/middleware/rateLimiter.ts` as tRPC middleware:
```typescript
import { TRPCError } from '@trpc/server';
import { middleware } from '../trpc.js';

const loginAttempts = new Map<string, { count: number; firstAttempt: number }>();
const MAX_ATTEMPTS = 5;
const WINDOW_MS = 15 * 60 * 1000; // 15 minutos

export const rateLimitLogin = middleware(async ({ ctx, next }) => {
  const ip = ctx.req.ip;
  const now = Date.now();
  const record = loginAttempts.get(ip);

  if (record) {
    if (now - record.firstAttempt > WINDOW_MS) {
      loginAttempts.delete(ip);
    } else if (record.count >= MAX_ATTEMPTS) {
      throw new TRPCError({
        code: 'TOO_MANY_REQUESTS',
        message: 'RATE_LIMITED',
      });
    }
  }

  const result = await next();

  // Increment on failed login (error thrown)
  // Note: successful logins don't increment — this is handled in the auth router
  return result;
});
```
